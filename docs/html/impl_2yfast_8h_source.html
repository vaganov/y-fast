218 Where insert(Leaf* leaf) {
219 auto pred = _trie.pred(leaf->key);
220 auto succ = (pred != nullptr) ? pred->nxt : _trie.leftmost();
221 XFastLeaf* xleaf;
222
223 if (pred != nullptr && succ != nullptr) {
224 if (_cmp(leaf->key, succ->value.leftmost()->key)) {
225 xleaf = pred;
226 }

228 xleaf = succ;

271 void remove(Leaf* leaf, XFastLeaf* hint = nullptr) {
272 auto xleaf = hint;
273 if (xleaf == nullptr || xleaf->value.find(leaf->key) != leaf) {
274 auto where = find(leaf->key);
275 xleaf = where.xleaf;
276 }
277 if (xleaf == nullptr) {
278 return;
279 }
280 xleaf->value.remove(leaf);
281 if (xleaf->value.size() < TREE_MERGE_THRESHOLD) {
282 auto neighbor = pick_neighbor(xleaf);
283 if (neighbor != nullptr) {
284 _trie.remove(xleaf);
285 _trie.remove(neighbor);
286 auto merged = Value::merge(std::move(xleaf->value), std::move(neighbor->value));
287 if (merged.size() > TREE_SPLIT_THRESHOLD) {
288 auto split_result = merged.split();
289 XFastLeaf* left = std::allocator_traits<Alloc>::allocate(_alloc, 1);
290 std::allocator_traits<Alloc>::construct(_alloc, left, split_result.left.root()->key, std::move(split_result.left));
291 XFastLeaf* right = std::allocator_traits<Alloc>::allocate(_alloc, 1);
292 std::allocator_traits<Alloc>::construct(_alloc, right, split_result.right.root()->key, std::move(split_result.right));
293 _trie.insert(left);
294 _trie.insert(right);
295 }
296 else {
297 XFastLeaf* merged_xleaf = std::allocator_traits<Alloc>::allocate(_alloc, 1);
298 std::allocator_traits<Alloc>::construct(_alloc, merged_xleaf, merged.root()->key, std::move(merged));
299 _trie.insert(merged_xleaf);
300 }
301 std::allocator_traits<Alloc>::destroy(_alloc, xleaf);
302 std::allocator_traits<Alloc>::deallocate(_alloc, xleaf, 1);
303 std::allocator_traits<Alloc>::destroy(_alloc, neighbor);
304 std::allocator_traits<Alloc>::deallocate(_alloc, neighbor, 1);
305 ++_rebuilds;
306 }
307 else if (xleaf->value.size() == 0) {
308 _trie.remove(xleaf);
309 std::allocator_traits<Alloc>::destroy(_alloc, xleaf);
310 std::allocator_traits<Alloc>::deallocate(_alloc, xleaf, 1);
311 ++_rebuilds;
312 }
313 }
314 else {
315 if (_cmp(xleaf->key, xleaf->value.leftmost()->key) || _cmp(xleaf->value.rightmost()->key, xleaf->key)) {
316 _trie.remove(xleaf);
317 XFastLeaf* reinserted_xleaf = std::allocator_traits<Alloc>::allocate(_alloc, 1);
318 std::allocator_traits<Alloc>::construct(_alloc, reinserted_xleaf, xleaf->value.root()->key, std::move(xleaf->value));
319 _trie.insert(reinserted_xleaf);
320 std::allocator_traits<Alloc>::destroy(_alloc, xleaf);
321 std::allocator_traits<Alloc>::deallocate(_alloc, xleaf, 1);
322 }
323 }

325 --_size;
326 }

345 static XFastLeaf* pick_neighbor(XFastLeaf* xleaf) {
346 auto pred = xleaf->prv;
347 auto succ = xleaf->nxt;
348 if (pred != nullptr && succ != nullptr) {
349 auto pred_merge_size = pred->value.size() + xleaf->value.size();
350 auto succ_merge_size = succ->value.size() + xleaf->value.size();
351 if (pred_merge_size <= TREE_SPLIT_THRESHOLD && succ_merge_size <= TREE_SPLIT_THRESHOLD) {
352 return pred_merge_size < succ_merge_size ? pred : succ;
353 }
354 if (pred_merge_size <= TREE_SPLIT_THRESHOLD) {
355 return pred;
356 }
357 if (succ_merge_size <= TREE_SPLIT_THRESHOLD) {
358 return succ;
359 }
360 return pred_merge_size > succ_merge_size ? pred : succ;
361 }
362 if (pred != nullptr) {
363 return pred;
364 }
365 if (succ != nullptr) {
366 return succ;
367 }
368 return nullptr;
369 }
